# 1. Create virtual environment:
#    python -m venv venv
#    source venv/bin/activate  # On Windows: venv\Scripts\activate
#
# 2. Install dependencies:
#    pip install -r requirements.txt
#
# 3. Install AgentField:
#    git clone https://github.com/Agent-Field/agentfield.git
#    cd agentfield
#    make install
#
# 4. Set environment variables:
#    export ANTHROPIC_API_KEY="your-api-key"
#    export AGENTFIELD_SERVER="http://localhost:8080"
#
# 5. Install language-specific tools:
#    - Go: go install github.com/golangci/golangci-lint/cmd/golangci-lint@latest

#    - Rust: cargo install cargo-audit
